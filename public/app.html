Como minerar

# 1. rode seu nó minerador
node bin/eav7.js mine --port 6071 \
  --peers http://127.0.0.1:6070

# 2. tenha EAV7 na carteira e faça stake
node bin/eav7.js stake \
  --wallet data/node-6071/validator-wallet.json \
  --amount 1000

# stake ≥ 100 zera taxas · ≥ 1000 = minerador (16 EAV7/bloco)
